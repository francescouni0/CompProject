            for name in files:

    for i, word in enumerate(r):
        
        if subdir in word:
            
            a.append(r[i])
            
    
    if "segmentation" in subdir:
        a.sort(key=lambda x: int(os.path.basename(x).split('_')[2]))
    
    else:
        a.sort(key=lambda x: int(os.path.basename(x).split('_')[3]))         
                    
                    
    return a
